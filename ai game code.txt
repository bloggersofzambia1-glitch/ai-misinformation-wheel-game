    // ------- Wheel Colors (repeat as necessary) ---------
    const SEGMENT_COLORS = [
      "#f39c12","#2ecc71","#3498db","#e67e22","#9b59b6","#16a085",
      "#d35400","#c0392b","#2980b9","#7f8c8d","#27ae60","#e84393"
    ];

    // ---------- Quiz game state -----------
    let usedIndices = [];
    let score = 0;
    let total = 0;
    let spinning = false;
    let selectedIndex = null; // For spinning wheel

    // ---------- DOM Elements -------------
    const wheel = document.getElementById("wheel");
    const spinBtn = document.getElementById("spin-btn");
    const wheelArea = document.getElementById("wheel-area");
    const scenarioTitle = document.getElementById("scenario-title");
    const questionText = document.getElementById("question");
    const optionsList = document.getElementById("options");
    const nextBtn = document.getElementById("next-btn");
    const scoreBar = document.getElementById("score");
    const questionSection = document.getElementById("question-section");
    const resultSection = document.getElementById("result-section");
    const finalScore = document.getElementById("final-score");
    const restartBtn = document.getElementById("restart-btn");

    // ---------- Wheel Drawing -------------
    function drawWheel(highlightSegment = null) {
      const ctx = wheel.getContext("2d");
      ctx.clearRect(0, 0, wheel.width, wheel.height);
      const n = SCENARIOS.length;
      const cx = wheel.width/2, cy = wheel.height/2, radius = wheel.width/2 - 6;
      const angleStep = 2 * Math.PI / n;

      for (let i = 0; i < n; i++) {
        ctx.beginPath();
        ctx.moveTo(cx, cy);
        ctx.arc(cx, cy, radius, i * angleStep, (i+1) * angleStep);
        ctx.fillStyle = (highlightSegment === i) ? "#fffac4" : SEGMENT_COLORS[i % SEGMENT_COLORS.length];
        ctx.fill();

        ctx.save();
        ctx.translate(cx, cy);
        ctx.rotate((i + 0.5) * angleStep);
        ctx.textAlign = 'right';
        ctx.font = "bold 15px var(--font)";
        ctx.fillStyle = "#fff";
        // Ellipsize scenario title
        let t = SCENARIOS[i].scenario;
        if (t.length > 17) t = t.slice(0,14) + "...";
        ctx.shadowColor = "#333a";
        ctx.shadowBlur = 4;
        ctx.fillText(t, radius-16, 8);
        ctx.restore();
      }
      // Outer border
      ctx.beginPath();
      ctx.arc(cx, cy, radius, 0, 2*Math.PI);
      ctx.lineWidth = 7;
      ctx.strokeStyle = "#fff";
      ctx.stroke();
    }

    drawWheel();

    // ----------- Spin Wheel Logic ------------
    let currentRotation = 0;

    function spinWheel() {
      if (spinning) return;
      if (usedIndices.length === SCENARIOS.length) {
        // End of game
        showResult();
        return;
      }
      spinning = true;
      spinBtn.disabled = true;
      drawWheel();

      // Pick a random unused segment index
      let idx;
      do {
        idx = Math.floor(Math.random() * SCENARIOS.length);
      } while (usedIndices.includes(idx));

      selectedIndex = idx;

      // Calculate the angle (in radians) where this segment's center lands at top (12 o'clock)
      const segments = SCENARIOS.length;
      const anglePer = 2*Math.PI / segments;
      const targetDeg = (270 - (idx * 360/segments)) % 360; // Top is 270°
      // Add several rounds to make it spin
      const extraRounds = Math.floor(Math.random()*2) + 4; // 4~5 wraps
      const finalDeg = 360*extraRounds + targetDeg;

      // Animate!
      let start = null;
      const duration = 3000 + Math.random()*500;
      let animFrame;

      function animate(ts) {
        if (!start) start = ts;
        let elapsed = ts - start;
        // Ease out cubic
        let t = Math.min(elapsed / duration, 1);
        let ease = (--t)*t*t+1; // cubic-out

        currentRotation = finalDeg * ease;

        wheel.style.transform = `rotate(${currentRotation}deg)`;
        if (elapsed < duration) {
          animFrame = requestAnimationFrame(animate);
        } else {
          // Snap to final
          wheel.style.transform = `rotate(${finalDeg}deg)`;
          spinning = false;
          usedIndices.push(selectedIndex);
          setTimeout(showQuestion, 400);
        }
      }
      animFrame = requestAnimationFrame(animate);
    }

    spinBtn.addEventListener("click", spinWheel);

    // ----------- Show Question Logic -------------
    function showQuestion() {
      const data = SCENARIOS[selectedIndex];
      scenarioTitle.textContent = "Scenario: " + data.scenario;
      questionText.textContent = data.question;
      optionsList.innerHTML = "";
      nextBtn.style.display = "none";
      questionSection.style.display = "block";
      spinBtn.style.display = "none";
      let answered = false;

      data.options.forEach((option, idx) => {
        const btn = document.createElement("button");
        btn.className = "option-btn";
        btn.textContent = String.fromCharCode(65+idx)+". "+option;
        btn.addEventListener("click", function() {
          if (answered) return;
          answered = true;
          total++;
          if (idx === data.correct) {
            btn.classList.add("correct");
            score++;
          } else {
            btn.classList.add("wrong");
          }
          // Reveal all: mark correct
          [...optionsList.children].forEach((b, i) => {
            b.disabled = true;
            if (i === data.correct) {
              b.classList.add("correct");
            } else if (i === idx) {
              b.classList.add(i === data.correct ? "correct" : "wrong");
            }
          });
          scoreBar.textContent = `${score} / ${total}`;
          nextBtn.style.display = "inline-block";
        });
        optionsList.appendChild(btn);
      });
    }

    nextBtn.addEventListener("click", () => {
      questionSection.style.display = "none";
      if (usedIndices.length === SCENARIOS.length) {
        showResult();
        return;
      }
      spinBtn.style.display = "inline-block";
      spinBtn.disabled = false;
      drawWheel();
    });

    // ------------- Show result logic -------------
    function showResult() {
      resultSection.style.display = "block";
      finalScore.textContent = `${score} / ${total}`;
      wheelArea.style.display = "none";
      questionSection.style.display = "none";
      spinBtn.style.display = "none";
      scoreBar.textContent = `${score} / ${total}`;
    }

    restartBtn.addEventListener("click", () => {
      usedIndices = [];
      score = 0;
      total = 0;
      spinning = false;
      selectedIndex = null;
      scoreBar.textContent = "0 / 0";
      wheelArea.style.display = "flex";
      questionSection.style.display = "none";
      resultSection.style.display = "none";
      spinBtn.style.display = "inline-block";
      spinBtn.disabled = false;
      wheel.style.transform = "rotate(0deg)";
      currentRotation = 0;
      drawWheel();
    });

    // -------- Keyboard: Space to spin/next ----------
    document.addEventListener("keydown", function(e) {
      if (e.key === " " && !spinning) {
        if (resultSection.style.display === "block") return;
        if (spinBtn.style.display !== "none" && !spinBtn.disabled) spinWheel();
        else if (nextBtn.style.display !== "none") nextBtn.click();
      }
    });
